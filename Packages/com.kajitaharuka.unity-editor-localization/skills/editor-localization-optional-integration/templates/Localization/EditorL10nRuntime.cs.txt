#if UNITY_EDITOR
namespace {{ROOT_NAMESPACE}}
{
    /// <summary>
    /// 使用中の<see cref="IEditorL10nBridge"/>を保持する差し替え点。
    /// 既定は基盤非依存の<see cref="DefaultEditorL10nBridge"/>。
    /// UnityEditorLocalizationが存在する場合のみ、連携assemblyが起動時に
    /// <see cref="SetBridge"/>で基盤連携実装へ差し替える。
    /// </summary>
    internal static class EditorL10nRuntime
    {
        private static IEditorL10nBridge _bridge = new DefaultEditorL10nBridge();

        internal static IEditorL10nBridge Bridge => _bridge;

        /// <summary>
        /// ブリッジ実装を差し替える。連携assemblyの起動処理からのみ呼ばれる想定。
        /// </summary>
        internal static void SetBridge(IEditorL10nBridge bridge)
        {
            if (bridge != null)
                _bridge = bridge;
        }
    }
}
#endif
