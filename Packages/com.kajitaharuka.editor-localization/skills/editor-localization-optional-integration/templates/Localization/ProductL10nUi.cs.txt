#if UNITY_EDITOR
using System;
using UnityEditor.UIElements;
using UnityEngine.UIElements;

namespace {{ROOT_NAMESPACE}}
{
    /// <summary>
    /// UI Toolkit要素を翻訳keyへ束ね、ロケール変更へ追従させるファサード。
    /// 実体は<see cref="EditorL10nRuntime.Bridge"/>へ委譲し、多言語基盤の有無を呼び出し側へ
    /// 意識させない。基盤が無い場合は単一言語で一度だけ文言を適用する。
    /// 旧来直接呼んでいた基盤の EditorL10nUi の代替として用いる。
    /// </summary>
    internal static class {{PREFIX}}L10nUi
    {
        private static IEditorL10nBridge Bridge => EditorL10nRuntime.Bridge;

        /// <summary>
        /// Labelのテキストを翻訳keyへ束ね、ロケール変更へ追従させる。
        /// </summary>
        internal static void BindText(Label label, string scope, string key, params object[] args)
        {
            if (label == null)
                return;

            void Apply() => label.text = Bridge.Tr(scope, key, args);
            Apply();
            Bridge.RegisterLocaleCallback(label, Apply);
        }

        /// <summary>
        /// Buttonのテキストとtooltipを翻訳keyへ束ね、ロケール変更へ追従させる。
        /// </summary>
        internal static void BindButton(Button button, string scope, string textKey, string tooltipKey = null, params object[] args)
        {
            if (button == null)
                return;

            void Apply()
            {
                button.text = Bridge.Tr(scope, textKey, args);
                if (!string.IsNullOrEmpty(tooltipKey))
                    button.tooltip = Bridge.Tr(scope, tooltipKey, Array.Empty<object>());
            }

            Apply();
            Bridge.RegisterLocaleCallback(button, Apply);
        }

        /// <summary>
        /// PropertyFieldのラベルとtooltipを翻訳keyへ束ね、ロケール変更へ追従させる。
        /// PropertyFieldはラベル要素が遅れて生成されるため、GeometryChangedでも再適用する。
        /// </summary>
        internal static void BindPropertyField(PropertyField field, string scope, string labelKey, string tooltipKey = null)
        {
            if (field == null)
                return;

            void Apply()
            {
                var labelElement = field.Q<Label>(className: BaseField<string>.labelUssClassName);
                if (labelElement != null)
                    labelElement.text = Bridge.Tr(scope, labelKey, Array.Empty<object>());
                if (!string.IsNullOrEmpty(tooltipKey))
                    field.tooltip = Bridge.Tr(scope, tooltipKey, Array.Empty<object>());
            }

            field.RegisterCallback<GeometryChangedEvent>(_ => Apply());
            Apply();
            Bridge.RegisterLocaleCallback(field, Apply);
        }

        /// <summary>
        /// ロケール変更時に再適用したいコールバックを要素へ紐づける。
        /// </summary>
        internal static void RegisterLocaleCallback(VisualElement element, Action callback)
        {
            Bridge.RegisterLocaleCallback(element, callback);
        }

        /// <summary>
        /// コンパクトな言語切替メニューを生成する。基盤が無い場合はnull（メニュー非表示）。
        /// </summary>
        internal static VisualElement CreateCompactLocaleMenu(string scope, string tooltipLabelKey)
        {
            return Bridge.CreateCompactLocaleMenu(scope, tooltipLabelKey);
        }
    }
}
#endif
